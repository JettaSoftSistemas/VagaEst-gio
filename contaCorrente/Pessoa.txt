public class Pessoa {
	
	private String nome;
	private int identidade;
	
	public Pessoa(String nome, int identidade) {
		super();
		this.nome = nome;
		this.identidade = identidade;
	}

	public String getNome() {
		return nome;
	}

	public int getIdentidade() {
		return identidade;
	}
	
}
